class Main {

    public static void pattern() {
        for (int i = 1; i <= 5; i++) {
            for (int j = 1; j <= 5; j++) {
                if (j == 2 || j == 3 || j == 4) {
                    System.out.print(" ");
                } else {
                    System.out.print("*");
                }
            }
            System.out.println(); // move to next line
        }
    }

    public static void main(String[] args) {
        pattern();
    }
}
